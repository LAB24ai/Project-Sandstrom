% Colors
\definecolor{primary}{RGB}{0, 82, 155}
\definecolor{secondary}{RGB}{100, 100, 100}

% Title formatting
\titleformat{\section}
    {\Large\bfseries\color{primary}}
    {\thesection}{1em}{}

    {\large\bfseries\color{secondary}}
    {\thesubsection}{1em}{}

    bgcolor=codebackground,
    fontsize=\small,
    frame=single,
    framesep=2mm,
    baselinestretch=1.2,
    linenos=true,
    breaklines=true,
    fontfamily=ttfamily
}

% Custom environments
\newenvironment{note}[1][]
    {\begin{tcolorbox}[
        colframe=primary,
        colback=white,
        arc=0mm,
        title=#1,
        fonttitle=\bfseries
    ]}
    {\end{tcolorbox}}

\newenvironment{warning}[1][]
    {\begin{tcolorbox}[
        colframe=red,
        colback=white,
        arc=0mm,
        title=#1,
        fonttitle=\bfseries
    ]}
    {\end{tcolorbox}}

% For better handling of figures
